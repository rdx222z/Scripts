local p=game.Players.LocalPlayer
local s=Instance.new("ScreenGui",p:WaitForChild("PlayerGui"))
s.Name="rdx222_UI" s.ResetOnSpawn=false s.ZIndexBehavior=Enum.ZIndexBehavior.Global
local f=Instance.new("Frame",s)
f.Size=UDim2.new(0,200,0,130)
f.Position=UDim2.new(0.5,-100,0.5,-65)
f.BackgroundColor3=Color3.fromRGB(30,30,30)
f.BorderSizePixel=0
f.Active=true
f.Draggable=true

local uic=Instance.new("UICorner",f)
uic.CornerRadius=UDim.new(0,12)

ub.Thickness=2
ub.Color=Color3.fromRGB(255,105,180)

local t=Instance.new("TextLabel",f)
t.Size=UDim2.new(1,0,0,25)
t.Position=UDim2.new(0,0,0,0)
t.BackgroundTransparency=1
t.Text="by rdx222"
t.TextColor3=Color3.fromRGB(255,255,255)
t.TextStrokeTransparency=0
t.Font=Enum.Font.GothamBlack
t.TextScaled=true

local toggle=Instance.new("TextButton",f)
toggle.Size=UDim2.new(0.9,0,0,30)
toggle.Position=UDim2.new(0.05,0,0.35,0)
toggle.BackgroundColor3=Color3.fromRGB(50,50,50)
toggle.Text="Steal: OFF"
toggle.TextColor3=Color3.fromRGB(255,255,255)
toggle.TextStrokeTransparency=0
toggle.Font=Enum.Font.GothamBold
toggle.TextScaled=true
Instance.new("UICorner",toggle).CornerRadius=UDim.new(0,8)
local togState=false

local label=Instance.new("TextLabel",f)
label.Position=UDim2.new(0.05,0,0.7,0)
label.Size=UDim2.new(0.9,0,0.2,0)
label.BackgroundTransparency=1
label.Text=""
label.TextColor3=Color3.fromRGB(255,255,255)
label.Font=Enum.Font.Gotham
label.TextScaled=true
label.TextStrokeTransparency=0

toggle.MouseButton1Click:Connect(function()
    togState=not togState
    toggle.Text="Steal: "..(togState and "ON" or "OFF")
    if togState then
        task.spawn(function()
            for i=1,3 do
                label.Text=i.." sec..."
                task.wait(1)
            end
            local c=game.Players.LocalPlayer.Character
            if c and c:FindFirstChild("HumanoidRootPart") then
                local startT=os.clock()
                local pos=CFrame.new(0,-500,0)
                while os.clock()-startT<1 do
                    c.HumanoidRootPart.CFrame=pos
                    task.wait()
                end
            end
            label.Text="Done!"
            task.wait(1)
            label.Text=""
            togState=false
            toggle.Text="Steal: OFF"
        end)
    end
end)
